  private static String[] getStdin() {
    Scanner scanner = new Scanner(System.in);
    ArrayList<String> lines = new ArrayList<>();
    while (scanner.hasNext()) {

    return lines.toArray(new String[lines.size()]);

  private static char[][] trimPiece(List<String> pieceLines) {
    int minX = pieceLines.size(), minY = pieceLines.get(0).length(), maxX = 0, maxY = 0;
    for (int i = 0; i < pieceLines.size(); i++) {
      String line = pieceLines.get(i);
      for (int j = 0; j < line.length(); j++) {
        if (line.charAt(j) == '#') {
          minX = Math.min(minX, i);
          maxX = Math.max(maxX, i);
          minY = Math.min(minY, j);
          maxY = Math.max(maxY, j);
        }
      }
    }

    char[][] piece = new char[maxX - minX + 1][maxY - minY + 1];
    for (int i = minX; i <= maxX; i++) {
      for (int j = minY; j <= maxY; j++) {
        piece[i - minX][j - minY] = pieceLines.get(i).charAt(j);
      }
    }

    return piece;
  }

  private static boolean dfs(int index, int N, List<char[][]> pieces, boolean[][] board, Set<String> visited) {
    if (index == pieces.size()) {
      for (int i = 0; i < N; i++) {
        for (int j = 0; j < N; j++) {
          if (!board[i][j]) return false;
        }
      }
      return true;
    }

    String boardKey = boardToString(board);
    if (visited.contains(boardKey)) return false;
    visited.add(boardKey);

    char[][] piece = pieces.get(index);

    for (int r = 0; r < 4; r++) {
      piece = rotateFunction(piece);
      for (int i = 0; i <= N - piece.length; i++) {
        for (int j = 0; j <= N - piece[0].length; j++) {
          if (isPossibleToPlace(piece, board, i, j, N)) {
            placeFunction(piece, board, i, j, true);
            if (dfs(index + 1, N, pieces, board, visited)) {
              return true;
            }
            placeFunction(piece, board, i, j, false);
          }
        }
      }
    }

    return false;
  }

  private static String boardToString(boolean[][] board) {
    StringBuilder sb = new StringBuilder();
    for (boolean[] row : board) {
      for (boolean cell : row) {
        sb.append(cell ? '1' : '0');
      }
    }
    return sb.toString();
  }

  private static char[][] rotateFunction(char[][] piece) {
    int sizeX = piece.length;
    int sizeY = piece[0].length;
    char[][] newPiece = new char[sizeY][sizeX];
    for (int i = 0; i < sizeX; i++) {
      for (int j = 0; j < sizeY; j++) {
        newPiece[j][sizeX - 1 - i] = piece[i][j];
      }
    }
    return newPiece;
  }

  private static void placeFunction(char[][] piece, boolean[][] board, int x, int y, boolean place) {
    int sizeX = piece.length;
    int sizeY = piece[0].length;
    for (int i = 0; i < sizeX; i++) {
      for (int j = 0; j < sizeY; j++) {
        if (piece[i][j] == '#') {
          board[x + i][y + j] = place;
        }
      }
    }
  }

  private static boolean isPossibleToPlace(char[][] piece, boolean[][] board, int x, int y, int N) {
    int sizeX = piece.length;
    int sizeY = piece[0].length;
    for (int i = 0; i < sizeX; i++) {
      for (int j = 0; j < sizeY; j++) {
        if (piece[i][j] == '#' && board[x + i][y + j]) {
          return false;
        }
      }
    }
    return true;
  }
